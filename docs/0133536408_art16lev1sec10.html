  c = [0 1];
  d = 0;
%

%
  alpha = exp(-0.1/lambda);
%
  tfinal = 5; % final simulation time
  sim('cd_IMC')
%
  figure(3)
  title('comparison of discrete and continuous IMC')
  subplot(2,1,1),plot(t,yimc,'--',t,yd,t,r,'-.')
  legend('continuous','discrete')
  xlabel('time, min')
  ylabel('y')
  subplot(2,1,2),plot(t,uimc,'--',t,ud)
  legend('continuous','discrete')
  ylabel('u')
  xlabel('time, min')
